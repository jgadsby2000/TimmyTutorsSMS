sqlite3 SADbois.db

CREATE TABLE Badges(
    	badgeID INTEGER 
	PRIMARY KEY,
	studentID INTEGER
);

CREATE TABLE BadgeColor(
	badgeColorID INTEGER
	PRIMARY KEY,
	badgeID INTEGER,
	badgeColor TEXT
);

CREATE TABLE Students(
	studentID INTEGER 
	PRIMARY KEY,
	firstname TEXT,
	lastname TEXT,
	dob DATE,
);

CREATE TABLE Group(
	groupID INTEGER
	PRIMARY KEY,
	studentID INTEGER
);

CREATE TABLE Teacher(
	teacherID INTEGER 
	PRIMARY KEY,
	firstname TEXT,
	lastname TEXT
);

CREATE TABLE Attendance(
	attendance INTEGER
	PRIMARY KEY,
	meetingID INTEGER,
	studentID INTEGER
);

CREATE TABLE Meeting(
	meetingID INTEGER
	PRIMARY KEY,
	teacherID INTEGER,
	time/date DATE,
	place TEXT,
	topic TEXT
);

CREATE TABLE WeekendIntensives(
	weekendIntensivesID INTEGER
	PRIMARY KEY,
	studentID INTEGER,
	week1 BOOLEAN DEFAULT FALSE,
	week2 BOOLEAN DEFAULT FALSE,
	week3 BOOLEAN DEFAULT FALSE,
	weekN BOOLEAN DEFAULT FALSE
);

CREATE TABLE WeeklyMeetings(
	weeklyMeetingID INTEGER
	PRIMARY KEY,
	studentID INTEGER,
	week1 BOOLEAN DEFAULT FALSE,
	week2 BOOLEAN DEFAULT FALSE,
	week3 BOOLEAN DEFAULT FALSE,
	weekN BOOLEAN DEFAULT FALSE
);

CREATE TABLE Leaderboard(
	leaderID INTEGER
	PRIMARY KEY,
	studentID INTEGER 
	rank INTEGER,
	score INTEGER
);

CREATE TABLE Test(
	testID INTEGER 
	PRIMARY KEY,
	testName Text,
	studentID INTEGER,
	badgeID INTEGER,
	completionStatus BOOLEAN
);

CREATE TABLE Topic(
	topicID INTEGER 
	PRIMARY KEY,
	studentID INTEGER,
	topicName TEXT,
	testID INTEGER,
	completionStatus BOOLEAN
);

CREATE TABLE Part(
	partID INTEGER 
	PRIMARY KEY,
	partName TEXT,
	topicID INTEGER,
	completionStatus BOOLEAN
);
